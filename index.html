    };

      ['menu-view', 'game-view', 'teacher-view', 'leaderboard-view', 'end-view'].forEach(v => document.getElementById(v).classList.add('hidden'));

      document.getElementById(viewId).classList.remove('hidden');

    };

    

    window.goToMenu = () => { window.stopTimer(); window.showView('menu-view'); };


    window.startGame = () => { 

      if (window.questions.length === 0) { 

        alert("Maglagay muna ng tanong sa Teacher Mode!"); 

        return; 

      }

      

      const pSection = document.getElementById('player-section').value.trim();

      const errBox = document.getElementById('menu-error');

      

      if (!pSection) {

        errBox.classList.remove('hidden');

        return;

      }

      errBox.classList.add('hidden');


      window.score = 0; 

      window.currentQIndex = 0; 

      window.questions.forEach(q => q.parts.forEach(p => p.isPlaced = false)); 

      

      window.showView('game-view'); 

      window.renderCurrentQuestion(); 

      window.startTimer();

    };


    window.endGame = () => {

      window.stopTimer();

      window.showView('end-view'); 

      document.getElementById('final-score-display').innerText = window.score; 

      document.getElementById('final-time-display').innerText = window.formatTime(window.timeSeconds);

      window.startConfetti();


      // Save to Firebase/Local

      const pSection = document.getElementById('player-section').value.trim();

      const record = {

        section: pSection.toUpperCase(),

        score: window.score,

        time: window.timeSeconds,

        date: new Date().toISOString()

      };


      if (window.db && window.user) {

        const colRef = collection(window.db, 'artifacts', appId, 'public', 'data', 'leaderboard');

        addDoc(colRef, record).catch(e => console.error(e));

      } else {

        window.leaderboardData.push(record);

      }

    };


    window.renderCurrentQuestion = () => {

      const q = window.questions[window.currentQIndex];

      if (!q) { window.endGame(); return; }

      

      document.getElementById('score-display').innerText = window.score;

      document.getElementById('progress-display').innerText = `Pangungusap: ${window.currentQIndex + 1} / ${window.questions.length}`;

      document.getElementById('placed-Simuno').innerHTML = ''; 

      document.getElementById('placed-Panaguri').innerHTML = '';

      

      const container = document.getElementById('draggables-container'); container.innerHTML = '';

      

      q.parts.forEach(part => {

        if (part.isPlaced) {

          const placeholder = document.createElement('div'); placeholder.className = "px-6 py-5 bg-indigo-50/20 text-indigo-300/40 text-2xl md:text-3xl font-black rounded-2xl border-4 border-dashed border-indigo-200/50 text-center select-none"; placeholder.innerText = part.text; container.appendChild(placeholder);

          window.addToZoneUI(part);

        } else {

          const dragItem = document.createElement('div'); dragItem.className = "px-6 py-5 bg-gradient-to-b from-white to-slate-50 text-2xl md:text-3xl font-black text-indigo-900 rounded-2xl border-b-8 border-r-4 border-indigo-200 cursor-grab active:cursor-grabbing shadow-xl text-center select-none touch-none"; dragItem.innerText = part.text;

          dragItem.addEventListener('pointerdown', (e) => window.onPointerDown(e, part, dragItem));

          container.appendChild(dragItem);

        }

      });

    };


    window.addToZoneUI = (part) => {

      const placedList = document.getElementById(`placed-${part.category}`);

      const block = document.createElement('div'); block.className = "bg-white px-4 py-3 rounded-2xl font-extrabold shadow-md border-3 border-emerald-400 text-slate-800 text-center w-full text-base md:text-lg flex items-center justify-center gap-2 animate-celebrate";

      block.innerHTML = `<span>${part.text}</span><span class="text-emerald-500 font-bold text-xl">✓</span>`;

      placedList.appendChild(block);

    };


    window.onPointerDown = (e, part, element) => {

      if (part.isPlaced) return; e.preventDefault();

      isDragging = true; draggedPartData = part; currentDragTargetElement = element;

      element.setPointerCapture(e.pointerId);

      

      const ghost = document.getElementById('drag-ghost'); 

      ghost.innerText = part.text; 

      ghost.style.left = `${e.clientX}px`; 

      ghost.style.top = `${e.clientY}px`; 

      ghost.classList.remove('hidden');

      

      element.classList.add('opacity-30');

      

      const onMove = (me) => { if (!isDragging) return; me.preventDefault(); ghost.style.left = `${me.clientX}px`; ghost.style.top = `${me.clientY}px`; };

      const onUp = (ue) => {

        element.releasePointerCapture(ue.pointerId); element.removeEventListener('pointermove', onMove); element.removeEventListener('pointerup', onUp);

        if (!isDragging) return; isDragging = false; ghost.classList.add('hidden'); element.classList.remove('opacity-30');

        const elementsBelow = document.elementsFromPoint(ue.clientX, ue.clientY); const dropZone = elementsBelow.find(el => el.getAttribute('data-category'));

        if (dropZone) window.handleDrop(dropZone.getAttribute('data-category'), ue.clientX, ue.clientY, draggedPartData);

      };

      

      element.addEventListener('pointermove', onMove); element.addEventListener('pointerup', onUp);

    };


    window.handleDrop = (category, x, y, part) => {

      if (part.category === category) {

        playTone('correct'); const pts = Math.floor(Math.random() * 5) + 1; window.score += pts; part.isPlaced = true;

        window.showPointsPopup(pts, x, y); window.renderCurrentQuestion();

        

        if (window.questions[window.currentQIndex].parts.every(p => p.isPlaced)) { 

          setTimeout(() => { window.currentQIndex++; window.renderCurrentQuestion(); }, 1200); 

        }

      } else {

        playTone('wrong'); const zone = document.getElementById(`zone-${category}`); if (zone) { zone.classList.add('animate-shake'); setTimeout(() => zone.classList.remove('animate-shake'), 500); }

      }

    };


    window.showPointsPopup = (pts, x, y) => {

      const c = document.getElementById('effect-container'), p = document.createElement('div');

      p.className = "fixed z-50 text-5xl text-yellow-500 animate-float-pop pointer-events-none font-black title-3d-yellow"; p.style.left = `${x}px`; p.style.top = `${y}px`; p.style.transform = 'translate(-50%, -50%)'; p.innerText = `+${pts} PUNTOS!`;

      c.appendChild(p); setTimeout(() => p.remove(), 1200);

    };


    window.startConfetti = () => {

      const c = document.getElementById('effect-container'); c.innerHTML = '';

      for (let i = 0; i < 30; i++) {

        const d = document.createElement('div'); d.className = `absolute rounded-full animate-float-pop bg-yellow-400 w-4 h-4`; d.style.left = `${Math.random() * 100}%`; d.style.top = `${Math.random() * 100}%`; d.style.animationDelay = `${Math.random() * 2}s`; d.style.animationIterationCount = 'infinite'; c.appendChild(d);

      }

    };


    // --- LEADERBOARD LOGIC ---

    window.showLeaderboard = () => {

      window.showView('leaderboard-view');

      window.renderLeaderboardUI();

    };


    window.renderLeaderboardUI = () => {

      const select = document.getElementById('section-filter');

      const list = document.getElementById('leaderboard-list');

      const emptyMsg = document.getElementById('leaderboard-empty');

      const currentFilter = select.value;


      // Extract unique sections

      const uniqueSections = [...new Set(window.leaderboardData.map(d => d.section))].filter(Boolean).sort();

      

      // Rebuild select options keeping current selection if possible

      select.innerHTML = '<option value="all">Lahat ng Seksyon</option>';

      uniqueSections.forEach(sec => {

        select.innerHTML += `<option value="${sec}">${sec}</option>`;

      });

      if (uniqueSections.includes(currentFilter) || currentFilter === 'all') {

        select.value = currentFilter;

      } else {

        select.value = 'all';

      }


      // Filter and Sort Data (Score DESC, Time ASC)

      let filteredData = [...window.leaderboardData];

      if (select.value !== 'all') {

        filteredData = filteredData.filter(d => d.section === select.value);

      }


      filteredData.sort((a, b) => {

        if (b.score !== a.score) return b.score - a.score;

        return a.time - b.time;

      });


      // Render

      list.innerHTML = '';

      if (filteredData.length === 0) {

        emptyMsg.classList.remove('hidden');

      } else {

        emptyMsg.classList.add('hidden');

        filteredData.forEach((row, idx) => {

          let rankMedal = `${idx + 1}`;

          if (idx === 0) rankMedal = '🥇';

          else if (idx === 1) rankMedal = '🥈';

          else if (idx === 2) rankMedal = '🥉';


          const tr = document.createElement('tr');

          tr.className = "border-b-2 border-slate-200 hover:bg-white transition-colors";

          tr.innerHTML = `

            <td class="p-4 text-center font-black text-2xl text-slate-500">${rankMedal}</td>

            <td class="p-4 font-bold text-lg text-slate-600">${row.section}</td>

            <td class="p-4 text-center font-black text-2xl text-amber-500">${row.score}</td>

            <td class="p-4 text-center font-bold text-lg text-emerald-600">${window.formatTime(row.time)}</td>

          `;

          list.appendChild(tr);

        });

      }

    };


    // --- TEACHER EDITOR LOGIC ---

    window.showTeacherMode = () => { window.showView('teacher-view'); window.renderTeacherQuestions(); };

    

    window.renderTeacherQuestions = () => {

      const content = document.getElementById('teacher-content'); content.innerHTML = '';

      if (window.editingQuestion) {

        const form = document.createElement('div'); form.className = "bg-amber-50 p-6 rounded-2xl border-4 border-amber-200 space-y-4";

        form.innerHTML = `

          <h3 class="font-extrabold text-2xl text-amber-900">${window.editingQuestion.id ? 'I-edit ang Pangungusap' : 'Bagong Pangungusap'}</h3>

          <div><label class="block font-black text-slate-700">Buong Pangungusap:</label><textarea id="edit-sentence" class="w-full p-3 rounded-xl border-4 border-amber-200 text-lg outline-none font-semibold" rows="2">${window.editingQuestion.sentence}</textarea></div>

          <div><label class="block font-black text-slate-700 mb-2">Hatiin ang Bahagi ng Pangungusap:</label><div id="parts-editor-container" class="space-y-2"></div><button onclick="window.addNewPartToEditor()" class="mt-3 text-blue-600 font-bold">+ Magdagdag ng Bahagi</button></div>

          <div class="mt-8 flex justify-end gap-3 border-t-2 pt-4"><button onclick="window.cancelEditing()" class="px-6 py-2 rounded-full bg-slate-200 font-black">Kanselahin</button><button onclick="window.saveEditing()" class="px-8 py-2 rounded-full text-white bg-green-500 font-black shadow-[0_4px_0_#15803d]">I-save</button></div>`;

        content.appendChild(form); window.renderPartsInsideEditor();

      } else {

        const listContainer = document.createElement('div'); listContainer.className = "space-y-4";

        const addBtn = document.createElement('button'); addBtn.className = "w-full py-5 border-4 border-dashed border-amber-300 text-amber-700 font-extrabold rounded-2xl text-2xl hover:bg-amber-50/50 transition-all"; addBtn.innerText = "+ Gumawa ng Bagong Tanong"; addBtn.onclick = () => { window.editingQuestion = { sentence: '', parts: [] }; window.renderTeacherQuestions(); };

        listContainer.appendChild(addBtn);

        window.questions.forEach((q, idx) => {

          const item = document.createElement('div'); item.className = "bg-white p-5 rounded-2xl shadow-lg flex flex-col md:flex-row justify-between items-center border-2 border-slate-100 gap-4";

          let badges = q.parts.map(p => `<span class="text-sm font-bold px-3 py-1 rounded-full border ${p.category === 'Simuno' ? 'bg-indigo-50 text-indigo-700' : 'bg-pink-50 text-pink-700'}">${p.text} (<b>${p.category}</b>)</span>`).join(' ');

          item.innerHTML = `<div class="w-full"><h4 class="font-extrabold text-xl text-slate-800 mb-2">${idx + 1}. ${q.sentence}</h4><div class="flex gap-2 flex-wrap">${badges}</div></div>

            <div class="flex gap-4"><button onclick="window.editQuestion(${q.id})" class="text-blue-500 hover:text-blue-700 font-black bg-blue-50 px-4 py-2 rounded-xl">I-edit</button><button onclick="window.removeQuestion(${q.id})" class="text-red-500 hover:text-red-700 font-black bg-red-50 px-4 py-2 rounded-xl">Burahin</button></div>`;

          listContainer.appendChild(item);

        });

        content.appendChild(listContainer);

      }

    };


    window.renderPartsInsideEditor = () => {

      const container = document.getElementById('parts-editor-container'); container.innerHTML = '';

      window.editingQuestion.parts.forEach((part, index) => {

        const item = document.createElement('div'); item.className = "flex gap-2 items-center bg-white p-3 rounded-xl border-2 border-amber-200";

        item.innerHTML = `<input type="text" value="${part.text}" onchange="window.updatePartText(${index}, this.value)" class="flex-1 p-2 border rounded-lg text-md font-semibold" placeholder="Bahagi ng pangungusap" />

          <select onchange="window.updatePartCategory(${index}, this.value)" class="p-2 border rounded-lg bg-white font-bold text-amber-900"><option value="Simuno" ${part.category === 'Simuno' ? 'selected' : ''}>Simuno</option><option value="Panaguri" ${part.category === 'Panaguri' ? 'selected' : ''}>Panaguri</option></select>

          <button onclick="window.removePartFromEditor(${index})" class="text-red-500 font-black px-2 text-xl hover:scale-110">✕</button>`;

        container.appendChild(item);

      });

    };


    window.addNewPartToEditor = () => { window.editingQuestion.parts.push({ id: Date.now().toString() + Math.random().toString(), text: '', category: 'Simuno', isPlaced: false }); window.renderPartsInsideEditor(); };

    window.removePartFromEditor = (idx) => { window.editingQuestion.parts.splice(idx, 1); window.renderPartsInsideEditor(); };

    window.updatePartText = (idx, val) => window.editingQuestion.parts[idx].text = val;

    window.updatePartCategory = (idx, val) => window.editingQuestion.parts[idx].category = val;

    window.editQuestion = (id) => { window.editingQuestion = JSON.parse(JSON.stringify(window.questions.find(q => q.id === id))); window.renderTeacherQuestions(); };

    window.removeQuestion = (id) => { window.questions = window.questions.filter(q => q.id !== id); window.renderTeacherQuestions(); };

    window.cancelEditing = () => { window.editingQuestion = null; window.renderTeacherQuestions(); };

    window.saveEditing = () => { 

      window.editingQuestion.sentence = document.getElementById('edit-sentence').value; 

      if (window.editingQuestion.id) { window.questions = window.questions.map(q => q.id === window.editingQuestion.id ? window.editingQuestion : q); } 

      else { window.editingQuestion.id = Date.now(); window.questions.push(window.editingQuestion); } 

      window.editingQuestion = null; window.renderTeacherQuestions(); 

    };

  </script>
